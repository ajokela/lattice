## Part 2 — LAT-442: harmonize channel-send eligibility, reject Ref/closure/iterator graphs

**The harmonized rule (one helper, three call sites):** a value may be sent iff
1. its phase is not `VTAG_SUBLIMATED`, and
2. it is not `VTAG_FLUID` — except scalars (`VAL_INT`, `VAL_FLOAT`, `VAL_BOOL`, `VAL_UNIT`), which are always copyable, and
3. its graph contains no `VAL_REF`, `VAL_CLOSURE`, or `VAL_ITERATOR` anywhere (regardless of phase — `value_detach`'s deep clone *shares* these kinds: same `LatRef` cell with non-atomic refcount, shared `captured_env`, shared iterator state; src/value.c:~285-295, ~346-381, ~508-519).

This is a relaxation for eval.c (unphased containers now allowed — they deep-detach safely, matching the default backend's current behavior), a relaxation for stackvm/regvm (fluid scalars now allowed, matching eval.c), and a tightening everywhere for shared-state kinds. The pinned test `test_channel_crystal_only_send` (fluid array rejected) stays green under this rule. This matches the move(x) design's eligibility matrix (design/channel-ownership-transfer.md §1: graphs containing REF/CLOSURE/ITERATOR are a hard error).

```c
/* LAT-442: kinds that value_detach SHARES rather than copies (non-atomic
 * refcount on the same LatRef cell, shared captured_env, shared iterator
 * state) must never cross a channel. Walk the graph looking for them. */
static const char *find_unsendable_kind(const LatValue *v) {
    switch (v->type) {
    case VAL_REF: return "Ref";
    case VAL_CLOSURE: return "closure";
    case VAL_ITERATOR: return "iterator";
    case VAL_ARRAY:
        for (size_t i = 0; i < v->as.array.len; i++) {
            const char *r = find_unsendable_kind(&v->as.array.elems[i]);
            if (r) return r;
        }
        return NULL;
    case VAL_TUPLE:
        for (size_t i = 0; i < v->as.tuple.len; i++) {
            const char *r = find_unsendable_kind(&v->as.tuple.elems[i]);
            if (r) return r;
        }
        return NULL;
    case VAL_STRUCT:
        for (size_t i = 0; i < v->as.strct.field_count; i++) {
            const char *r = find_unsendable_kind(&v->as.strct.field_values[i]);
            if (r) return r;
        }
        return NULL;
    case VAL_ENUM:
        for (size_t i = 0; i < v->as.enm.payload_count; i++) {
            const char *r = find_unsendable_kind(&v->as.enm.payload[i]);
            if (r) return r;
        }
        return NULL;
    case VAL_MAP:
        for (size_t i = 0; i < v->as.map.map->cap; i++) {
            if (v->as.map.map->entries[i].state == MAP_OCCUPIED) {
                const char *r = find_unsendable_kind((const LatValue *)v->as.map.map->entries[i].value);
                if (r) return r;
            }
        }
        return NULL;
    case VAL_SET:
        for (size_t i = 0; i < v->as.set.map->cap; i++) {
            if (v->as.set.map->entries[i].state == MAP_OCCUPIED) {
                const char *r = find_unsendable_kind((const LatValue *)v->as.set.map->entries[i].value);
                if (r) return r;
            }
        }
        return NULL;
    default: return NULL;
    }
}

const char *value_send_ineligible(const LatValue *v) {
    bool scalar = v->type == VAL_INT || v->type == VAL_FLOAT || v->type == VAL_BOOL || v->type == VAL_UNIT;
    if (v->phase == VTAG_SUBLIMATED) return "cannot send a sublimated value on a channel";
    if (v->phase == VTAG_FLUID && !scalar) return "can only send crystal (frozen) values on a channel";
    const char *kind = find_unsendable_kind(v);
    if (kind) {
        if (kind[0] == 'R') return "cannot send a value containing a Ref on a channel";
        if (kind[0] == 'c') return "cannot send a value containing a closure on a channel";
        return "cannot send a value containing an iterator on a channel";
    }
    return NULL;
}
```

## Part 3 — LAT-440: nested index assignment in the self-hosted compiler

`compiler/latc.lat` is single-pass (no AST), so it can't re-walk an expression tree like the C compiler's `emit_index_write_back` (src/stackcompiler.c:2256-2294). Instead: scan the bracket groups of the target first (token positions only, no emission), then emit in the required stack order by **re-parsing each index span from its recorded token position**. The C compiler also re-evaluates index expressions during write-back, so semantics match.

Bytecode shape for `g[i][j] = v` (matching the C compiler):
```
<v>  <get g> <i> OP_INDEX  <j>  OP_SET_INDEX     # [v, g[i], j] → [g[i]']
<i>  OP_SET_INDEX_LOCAL slot(g)                  # root write-back (local)
     — or —  <i> OP_SET_INDEX_GLOBAL gcidx       # (global)
```

### Task 6: implement nested index targets in latc.lat

**Files:**
- Modify: `compiler/latc.lat` (helpers near the other parse utilities ~line 4030; dispatch inside `parse_compound_assignment_target` ~line 4176)

**Language constraints reminder (latc.lat must compile under BOTH the C compiler and itself):** typed params (`: any`), no map literals, `${...}` interpolation, `test`/`anneal` are keywords, empty closure is `| |`, no chained calls (`f()(x)`), arrays mutate in place via `.push()` and `a[i] = v`.

**Token facts:** punctuation tokens have `.type` equal to the symbol itself (existing code checks `tokens[pos + 1].type == "+"`), so `tokens[i].type == "["` is the correct bracket check and is immune to `"["` inside string literals.

- [ ] **Step 1: Add helper functions before `parse_assignment` (~line 4030)**

```lattice
// LAT-440: nested index assignment support ------------------------------

// Token index of the ']' matching the '[' at tokens[start].
fn find_matching_bracket(start: any) {
    flux depth = 0
    flux i = start
    while i < len(tokens) {
        let tt = tokens[i].type
        if tt == "[" {
            depth = depth + 1
        }
        if tt == "]" {
            depth = depth - 1
            if depth == 0 {
                return i
            }
        }
        i = i + 1
    }
    compiler_error("unmatched '[' in index expression", current_line())
    return -1
}

// From a '[' at tokens[start], count consecutive [..] groups. Returns the
// group count (>= 2) only when the chain is a nested assignment target
// (ends at an assignment operator); otherwise 0 so the caller falls
// through to the existing single-level path.
fn index_chain_depth(start: any) {
    flux i = start
    flux depth = 0
    while i < len(tokens) && tokens[i].type == "[" {
        let close = find_matching_bracket(i)
        depth = depth + 1
        i = close + 1
    }
    if depth < 2 {
        return 0
    }
    let t = tokens[i].type
    if t == "=" || t == "+=" || t == "-=" || t == "*=" || t == "/=" || t == "%=" {
        return depth
    }
    return 0
}

// Re-emit the index expression whose first token is at index start.
// Saves and restores pos, so it can be called out of stream order.
fn emit_index_span(start: any) {
    let saved = pos
    pos = start
    parse_expression()
    pos = saved
}
```

- [ ] **Step 2: Add the nested-target compiler**

```lattice
// Compile a nested index assignment g[i0][i1]..[iN-1] OP= value (depth >= 2).
// Mirrors the C compiler's emit_index_write_back (stackcompiler.c): assign at
// the innermost level, then write each modified intermediate back up to the
// root variable. Index expressions are re-emitted from their token spans, so
// (exactly like the C compiler) they are evaluated more than once.
fn parse_nested_index_assignment(name: any, depth: any, line: any) {
    // Record the first inner token of each [..] group; leave pos at the operator.
    flux spans = []
    flux i = pos
    flux k = 0
    while k < depth {
        spans.push(i + 1)
        i = find_matching_bracket(i) + 1
        k = k + 1
    }
    pos = i
    let op = advance().text

    if op == "=" {
        parse_expression()                          // [v]
    } else {
        // Compound: current value, then apply the operator to the rhs.
        emit_variable_get(name, line)               // [g]
        k = 0
        while k < depth {
            emit_index_span(spans[k])
            emit_byte(OP_INDEX, line)
            k = k + 1
        }                                           // [cur]
        parse_expression()                          // [cur, rhs]
        match op {
            "+=" => emit_byte(OP_ADD, line),
            "-=" => emit_byte(OP_SUB, line),
            "*=" => emit_byte(OP_MUL, line),
            "/=" => emit_byte(OP_DIV, line),
            _ => emit_byte(OP_MOD, line)
        }                                           // [v]
    }

    // Innermost assignment: [v] -> [modified innermost container]
```
